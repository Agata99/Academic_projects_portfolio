Utworzenie funkcji generujƒÖcej liczby pseudolosowe z rozk≈Çadu normalnego przy u≈ºyciu rozk≈Çadu jednostajnego na odc [0,1] z wykorzystaniem Centralnego Twierdzenia Granicznego.

CTG <- function(N,M,s,w){
  if(w<0){print("Wariancja musi byƒá liczbƒÖ dodatniƒÖ")}
  else{
  x={};
  z={};
  for(k in 1:N){
    x <- runif(M,min=0,max=1) #losujemy M liczb z prz[0,1] rozk≈Çadu jednostajnego
    suma=sum(x)
    
    z[k]=sqrt(w)*(suma-M*1/2)/(sqrt((1/12)*M)) + s;
  
    
  }
  return(z);
  }}

BM <- function(N,s,w){
  if(w<0){print("Wariancja musi byƒá liczbƒÖ dodatniƒÖ")}
  else{
  z={};
  a={};
  v={};
  if(N%%2==0){N2 = N/2}
  else(N2 = floor(N/2)+1)
  for(i in 1:N2){
    u1 = runif(1,0,1)  #losujemy 2 liczby z prz[0,1] rozk≈Çadu jednostajnego
    u2 = runif(1,0,1)
    fi = 2*(pi)*u1
    R = sqrt(-2*log(u2))
    x = sqrt(w)*R*cos(fi) + s
    y = sqrt(w)*R*sin(fi) + s
  
     z[i]=x
     a[i]=y
  }
  if(N%%2==0){
  v=c(z,a)
  
  return(v)
  }
  if(N==1){
    return(z)
    }
  else{
    v=c(z,a[1:(N2 - 1)])
    
    return(v)
  }
  }
}

procent <- function(x,a,b){
  
  proc <- (length(x[x>=a & x<= b])/length(x))*100
  return(proc)
}
